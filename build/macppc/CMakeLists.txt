cmake_minimum_required(VERSION 3.10)
project(RelicMacPPC C CXX)

set(ROOT ${CMAKE_CURRENT_SOURCE_DIR}/../..)
set(SRC  ${ROOT}/src)
set(BEARSSL_ROOT ${ROOT}/third_party/bearssl)

# ---------------------------------------------------------------------------
# BearSSL static library (all sources -- pruning happens later)
# ---------------------------------------------------------------------------
file(GLOB BEARSSL_SRC
    ${BEARSSL_ROOT}/src/settings.c
    ${BEARSSL_ROOT}/src/aead/*.c
    ${BEARSSL_ROOT}/src/codec/*.c
    ${BEARSSL_ROOT}/src/ec/*.c
    ${BEARSSL_ROOT}/src/hash/*.c
    ${BEARSSL_ROOT}/src/int/*.c
    ${BEARSSL_ROOT}/src/kdf/*.c
    ${BEARSSL_ROOT}/src/mac/*.c
    ${BEARSSL_ROOT}/src/rand/*.c
    ${BEARSSL_ROOT}/src/rsa/*.c
    ${BEARSSL_ROOT}/src/ssl/*.c
    ${BEARSSL_ROOT}/src/symcipher/*.c
    ${BEARSSL_ROOT}/src/x509/*.c
)
add_library(bearssl STATIC ${BEARSSL_SRC})
target_include_directories(bearssl PUBLIC ${BEARSSL_ROOT}/inc PRIVATE ${BEARSSL_ROOT}/src)
target_compile_definitions(bearssl PRIVATE BR_USE_URANDOM=0 BR_USE_WIN32_RAND=0 BR_POWER8=0)
target_compile_options(bearssl PRIVATE -O2)

# ---------------------------------------------------------------------------
# Open Transport static glue (Apple SDK; not redistributed -- see README)
# ---------------------------------------------------------------------------
set(OTGLUE_DIR ${ROOT}/third_party/otglue)
if(NOT EXISTS ${OTGLUE_DIR}/OpenTransportAppPPC.o)
    message(FATAL_ERROR
        "Missing ${OTGLUE_DIR}/OpenTransportAppPPC.o\n"
        "Run tools/fetch_otglue.sh on the host first.")
endif()
add_library(otglue STATIC IMPORTED)
set_target_properties(otglue PROPERTIES IMPORTED_LOCATION
    ${CMAKE_CURRENT_BINARY_DIR}/libotglue.a)
add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/libotglue.a
    COMMAND ${CMAKE_AR} crs ${CMAKE_CURRENT_BINARY_DIR}/libotglue.a
            ${OTGLUE_DIR}/OpenTransportAppPPC.o
            ${OTGLUE_DIR}/OpenTptInetPPC.o
    DEPENDS ${OTGLUE_DIR}/OpenTransportAppPPC.o
            ${OTGLUE_DIR}/OpenTptInetPPC.o)
add_custom_target(otglue_ar DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/libotglue.a)
add_dependencies(otglue otglue_ar)

# ---------------------------------------------------------------------------
# Relic
# ---------------------------------------------------------------------------
file(GLOB CORE_SRC ${SRC}/core/*.c)
# icons.r (generated by tools/gen_macppc_icons.py) carries the Finder icon
# family + BNDL; CREATOR must match the signature resource inside it.
add_application(Relic
    ${SRC}/main.c
    ${CORE_SRC}
    ${SRC}/plat/plat_cfg.c
    ${SRC}/plat/macppc/plat_macppc.c
    ${SRC}/plat/macppc/size.r
    ${SRC}/plat/macppc/icons.r
    CREATOR "RLiC"
    CONSOLE)
target_include_directories(Relic PRIVATE ${BEARSSL_ROOT}/inc ${SRC} ${SRC}/plat/macppc)
target_compile_definitions(Relic PRIVATE
    BR_USE_URANDOM=0 BR_USE_WIN32_RAND=0 BR_USE_UNIX_TIME=0
    BUILD_UNIX_DAYS=${BUILD_UNIX_DAYS})
target_link_libraries(Relic bearssl otglue OpenTransportLib OpenTptInternetLib
    AppleScriptLib)
